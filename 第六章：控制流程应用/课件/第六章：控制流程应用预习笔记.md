while True:
    gamelist = ["石头", "剪刀", "布"]

    # 判断人输的情况：
    elif (user=="石头" and com=="布") or (user=="剪刀" and com=="石头") or (user=="布" and com=="剪刀") :
        print("恭喜电脑赢了",f"电脑出：{com} 你出：{user}")

        print("平局！")

        break    # 结束循环

在执行 while 循环或者 for 循环时，只要循环条件满足，程序将会一直执行循环体，不停地转圈。

"""
# 检查 1-10中有没有既能够被2整除，又能被3整除的数字，

    if i%2==0 and i%3==0:
        continue    # 跳过本次循环
    print(i)
