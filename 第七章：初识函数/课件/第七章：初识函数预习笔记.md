```python
"""
一、什么是函数？
概念：在编程中，由程序员设计制作好的一个可以重复利用的代码块

二、函数怎么写？
函数语法结构：
def 函数名(参数列表):
    函数代码块
    return 值
注意：1、def 关键词和 函数名 是必须要有的
     2、函数名字，即函数标识符，需要符合标识符命名规则，
     3、参数列表 和 return 返回值可以有可以没有

三、函数创建好后如何使用？
调用函数语法： 函数名(参数)
注意：
1、函数创建了之后，必须通过函数名去进行调用
2、函数在使用的时候，是否要传参，以及是否要接受返回值，是根据函数的设定来的
3、如果函数在创建的时候，写了参数列表，写了返回值，那么就是要传参的，也是要接受返回值的，反之则不用
"""

# 案例 ，假设以生活中的事务举例
# 榨汁机举例
# # 定义了一个函数  ，函数名字 zhaZhiJi ，函数参数(行参) suiGuo
def zhaZhiJi(suiGuo):
    print("正在榨果汁！")
    x = f"{suiGuo}汁"     #  生成了一个字符串，水果汁的字符串
    return x      # 返回了果汁

# 调用函数时（） 中的"苹果" 是实参
beiZi1 = zhaZhiJi("苹果")     # 调用了我做好的函数工具，然后传了一个参数进去
beiZi2 = zhaZhiJi("胡萝卜")
beiZi3 = zhaZhiJi("美国")
print(beiZi1)
print(beiZi2)
print(beiZi3)

# 根据写法不同
# 1、无参数无返回值函数
# 当我们在设计函数工具的时候，我不需要它有参数，也不需要它有返回值，
# 用户只需要叫唤一下这函数的名字就能够实现特定的效果
# 而不用去关注到底要传什么参，倒是要用什么变量去接收返回值
# 假设以心灵鸡汤举例
def func1():
    print("刚刚喝了个心灵鸡汤，感觉没啥用")
func1()

# 应用举例：
def func2():
    for i in range(1,10,1):
        for x in range(1, i + 1):
            print(f"{x}*{i}={i*x}",end="\t")
        print()  # 整个内层循环结束之后再进行换行
func2()

# 2、有参数无返回值函数
# 假设用生活中的空调举例
def func3(weiDu,fenSu):
    print(f"正在开启空调，温度为{weiDu}摄氏度，风速为{fenSu} ！")
y = func3(26,"高")
print(y)

# 应用举例：
li = [1,2,3,4,5,6]     # 列表可变数据类型
x = li.append("hello")    # 追加
print(x)
print(li)     # 查看列表源数据

# 3、无参数有返回值函数
# 假设以抽奖举例
def func4():
    import random      # random 是python内置抽奖工具库
    # 用到random工具库中的一个随机序列值的方法
    jiangPing = ["一等奖","二等奖","三等奖"]
    x = random.choice(jiangPing)
    y = random.choice(jiangPing)
    z = random.choice(jiangPing)
    return x,y,z      # renturn 只能返回一个元素，这个元素里面可以包含多个值
print(func4())
# 应用举例:
st = "Python1024"
st2 = st.upper()   # 转大写
print(st2)
print(st)

# 4、有参数有返回值函数
# 假设以两位数加法计算为例
def add(a,b):
    c = a+b
    return c
c = add(1,1)
print(c)
# 应用举例:
name = input("请输入名字>>")
print(name)


# 函数参数：
# 1、必备参数(位置参数、形参)
def func6(a,b,c):
    print(a,b,c)
func6(1,"hello",3)

# 2、默认参数
def func7(name,passWord="1024"):
    print(name,passWord)
func7("不渝")
func7("小柒",1106)

# 覆盖print源码中的sep值
print("今天","天气真好",sep="+++&77&&777+++")

# 在python中print打印不换行,是在()末尾加一个end=""

# 3、不定长参数
# 使用场景：设计函数者不确定使用者会需要传多少个参数的时候，可以使用位置不定长参数，实现灵活传参。
# 位置不定长参数 ==> *args
# 将所有无形参认领的实参都整合到一起,保存为元组,放到位置不定长参数中
def func8(name,passWord="1024",*args):
    print(name,passWord,args)
func8("不渝","1106",1,2,3,4,5,6,7)

# 关键字不定长参数 ==》 **kwargs
def func9(name,passWord="1024",*args,**kwargs):
    a = kwargs
    print(a.keys(),a.values(),a.items())
    print(name,passWord,args,kwargs)
func9("不渝","1106",1,2,3,4,5,6,url="xxxx",head="yyyyy",bbb="bbb")
input()
```
